zambian_drc_stability_relationship = {
	enable = {
		tag = ZAM
		country_exists = DRC
		OR = {
			country_exists = MLC
			country_exists = RCD
		}
		DRC = { has_democratic_form_of_government = yes }
	}
	abort = {
		OR = {
			NOT = { country_exists = DRC }
			DRC = { has_democratic_form_of_government = no }
			NOT = {
				country_exists = MLC
				country_exists = RCD
			}
		}
	}
	
	ai_strategy = {
		type = protect
		id = "DRC"
		value = 200
	}
}
